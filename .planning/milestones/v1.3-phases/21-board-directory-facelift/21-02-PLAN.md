          subscribed?: true,
          required_subscription?: false,
          unread_count: 3,
          last_post_at: ten_min_ago()

            category: %{id: "c1", name: category_name},
            boards: [Map.merge(defaults, board_attrs)]
          }

          tree = BoardTree.render(state, theme: theme)
          assert flatten_text(tree) != ""
