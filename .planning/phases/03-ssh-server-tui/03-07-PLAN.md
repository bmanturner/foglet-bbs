```elixir
@wrap_width 50

def render(%{message: msg} = spec) do
  type = Map.get(spec, :type, :info)
  title = Map.get(spec, :title, title_for(type))
  color = color_for(type)

  wrapped_lines =
    msg
    |> word_wrap(@wrap_width)
    |> Enum.map(fn line -> text(line, fg: color) end)

  column [] do
    [text(" #{title} ", style: [:bold]), divider()] ++
      wrapped_lines ++
      [text(key_hint_for(type), style: [:dim])]
  end
end

# Wrap a string to <= max_width columns on whitespace boundaries.
# Words longer than max_width are placed on their own line (not split mid-word).
defp word_wrap(text, max_width) when is_binary(text) and is_integer(max_width) do
  text
  |> String.split(~r/\s+/, trim: true)
  |> Enum.reduce([""], fn word, [current | rest] ->
    cond do
      current == "" -> [word | rest]
      String.length(current) + 1 + String.length(word) <= max_width -> ["#{current} #{word}" | rest]
      true -> [word, current | rest]
    end
  end)
  |> Enum.reverse()
end

defp key_hint_for(:confirm), do: "[Y] Yes   [N] No"
defp key_hint_for(_), do: "[Enter] OK"
```

From lib/foglet_bbs/markdown.ex — current output is an ANSI-escaped binary.
The TEXT_TO_STYLE mapping must be preserved but the return shape changes from
`String.t()` to `[{String.t(), style_atom()}]` where
`style_atom() :: :plain | :bold | :italic | :dim | :underline`.

Proposed structure (Gap 5):
```elixir
@type style_atom :: :plain | :bold | :italic | :dim | :underline
@type rendered :: [{String.t(), style_atom()}]

@spec render(String.t()) :: rendered()
def render(markdown) when is_binary(markdown) do
  sanitized = strip_ansi(markdown)
  html = MDEx.to_html!(sanitized)
  html_to_tuples(html)
end
```

`html_to_tuples/1` walks the HTML and emits `{text, style}` tuples. Each
paragraph, list item, heading, code block etc. contributes one or more tuples,
including `{"\n", :plain}` between block-level elements so downstream rendering
produces multi-line output. Implementation note: because MDEx emits nested
tags, the simplest correct approach is a small recursive-descent parser using
Floki (already in deps — MDEx parses and re-emits HTML). If Floki is not a
dep yet, add it (`{:floki, "~> 0.36"}`). Otherwise, keep using the regex
pipeline but have each substitution emit a marker like `{BOLD}text{/BOLD}`,
then split on markers in a second pass.

**Required output shape** — all existing test cases in
`test/foglet_bbs/markdown_test.exs` must be updated:

| Input                     | Output tuples                                              |
|---------------------------|------------------------------------------------------------|
| `"**hello**"`             | `[{"hello", :bold}]`                                       |
| `"*hi*"`                  | `[{"hi", :italic}]`                                        |
| `` "`code`" ``            | `[{"code", :dim}]`                                         |
| `"# Title"`               | `[{"TITLE", :underline}, {"\n", :plain}]`                  |
| `"plain text"`            | `[{"plain text", :plain}, {"\n", :plain}]`                 |
| `"**a** and *b*"`         | `[{"a", :bold}, {" and ", :plain}, {"b", :italic}, {"\n", :plain}]` |
| `"[click](https://x)"`    | `[{"click (https://x)", :plain}, {"\n", :plain}]`          |

(Trailing `{"\n", :plain}` on paragraph/heading blocks matches the existing
behaviour where render/1 appends block newlines.)

Helper needed in BOTH PostComposer and PostReader — extract into each module

```elixir
# Walks a [{text, style}] list from Foglet.Markdown.render/1 and produces
# a column of text/2 elements styled appropriately. Stays inline (no block
# rendering of its own).
defp render_markdown_tuples(tuples) when is_list(tuples) do
  column style: %{gap: 0} do
    Enum.map(tuples, fn
      {s, :bold} -> text(s, style: [:bold], fg: :green)
      {s, :italic} -> text(s, style: [:italic], fg: :green)
      {s, :dim} -> text(s, style: [:dim], fg: :green)
      {s, :underline} -> text(s, style: [:underline], fg: :green)
      {s, :plain} -> text(s, fg: :green)
      {s, _} -> text(s, fg: :green)
    end)
  end
end
```
