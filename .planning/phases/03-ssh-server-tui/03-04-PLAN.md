    ```elixir
    defmodule Foglet.TUI.Screens.MainMenu do
      @moduledoc "BBS main menu — primary screen after login (SSH-07, SSH-08)."

      alias Foglet.TUI.Widgets.{KeyBar, StatusBar}
      import Raxol.View.Elements

      @menu_items [
        {"B", "Browse Boards", :board_list},
        {"C", "Compose New Thread", :post_composer},
        {"P", "Profile", :main_menu},
        {"Q", "Logout", :logout}
      ]

      @spec render(map()) :: any()
      def render(state) do
        handle = state.current_user && state.current_user.handle

        panel title: "Foglet BBS", border: :single do
          StatusBar.render(%{handle: handle, location: "Main Menu"})

          box style: [flex_direction: :column, padding: 2] do
            text(content: "Welcome back, #{handle || "guest"}.", color: :green)
            text(content: "")

            for {k, label, _target} <- @menu_items do
              text(content: "  [#{k}] #{label}", color: :green)
            end
          end

          KeyBar.render([{"B", "Boards"}, {"C", "Compose"}, {"Q", "Logout"}])
        end
      end

      @spec handle_key(map(), map()) :: {:update, map(), list()} | :no_match
      def handle_key(%{key: key}, state) when key in ["b", "B"] do
        {:update, %{state | current_screen: :board_list}, [{:load_boards}]}
      end

      def handle_key(%{key: key}, state) when key in ["c", "C"] do
        # Compose a NEW thread — requires board selection first; for now, just open composer
        # with no thread context. Composer checks state.current_thread for reply vs new-thread mode.
        {:update, %{state | current_screen: :post_composer, composer_draft: "", current_thread: nil}, []}
      end

      def handle_key(%{key: key}, state) when key in ["q", "Q"] do
        {:update, state, [{:terminate, :logout}]}
      end

      def handle_key(_key, _state), do: :no_match
    end
    ```

    ```elixir
    defmodule Foglet.TUI.Screens.BoardList do
      @moduledoc """
      Subscribed-boards list (SSH-07, SSH-08).

      Uses a domain-adapter pattern so tests can inject a fake Foglet.Boards.
      state.screen_state[:board_list] holds {:selected_index}.
      """

      alias Foglet.TUI.Widgets.{KeyBar, StatusBar}
      import Raxol.View.Elements

      @spec render(map()) :: any()
      def render(state) do
        ss = get_in(state.screen_state, [:board_list]) || %{selected_index: 0}
        boards = state.board_list || []

        panel title: "Boards", border: :single do
          StatusBar.render(%{
            handle: state.current_user && state.current_user.handle,
            location: "Boards"
          })

          cond do
            boards == [] and state.board_list == [] ->
              text(content: "No boards subscribed. Ask your sysop to subscribe you.", color: :yellow)

            boards == [] ->
              text(content: "Loading...", color: :bright_black)

            true ->
              box style: [flex_direction: :column, padding: 1] do
                for {board, idx} <- Enum.with_index(boards) do
                  marker = if idx == ss.selected_index, do: "▶ ", else: "  "
                  unread = board_unread(board)
                  unread_str = if unread > 0, do: " (#{unread} unread)", else: ""

                  text(
                    content: "#{marker}#{board.name}#{unread_str}",
                    color: if(idx == ss.selected_index, do: :bright_green, else: :green)
                  )
                end
              end
          end

          KeyBar.render([
            {"↑/↓", "Select"},
            {"Enter", "Open"},
            {"Q", "Back"}
          ])
        end
      end

      @spec handle_key(map(), map()) :: {:update, map(), list()} | :no_match
      def handle_key(%{key: "j"}, state), do: move_selection(state, +1)
      def handle_key(%{key: "down"}, state), do: move_selection(state, +1)
      def handle_key(%{key: "k"}, state), do: move_selection(state, -1)
      def handle_key(%{key: "up"}, state), do: move_selection(state, -1)

      def handle_key(%{key: "enter"}, state) do
        boards = state.board_list || []
        ss = get_in(state.screen_state, [:board_list]) || %{selected_index: 0}

        case Enum.at(boards, ss.selected_index) do
          nil ->
            :no_match

          board ->
            new_state = %{
              state
              | current_board: board,
                current_screen: :thread_list,
                screen_state: Map.put(state.screen_state, :thread_list, %{selected_index: 0})
            }

            {:update, new_state, [{:load_threads, board.id}]}
        end
      end

      def handle_key(%{key: key}, state) when key in ["q", "Q"] do
        {:update, %{state | current_screen: :main_menu}, []}
      end

      def handle_key(_key, _state), do: :no_match

      @doc """
      Called by App.update/2 in response to a {:load_boards} command.
      Populates state.board_list by calling the configured domain module.
      """
      @spec load_boards(map()) :: {map(), list()}
      def load_boards(state) do
        boards_mod = domain_module(state, :boards)

        cond do
          function_exported?(boards_mod, :list_subscribed_boards, 1) ->
            boards = boards_mod.list_subscribed_boards(state.current_user)
            {%{state | board_list: boards}, []}

          true ->
            # Phase 2 not executed — leave empty so render shows the "No boards" message
            {%{state | board_list: []}, []}
        end
      end

      # --- Private ---

      defp move_selection(state, delta) do
        boards = state.board_list || []

        if boards == [] do
          :no_match
        else
          ss = get_in(state.screen_state, [:board_list]) || %{selected_index: 0}
          new_idx = (ss.selected_index + delta) |> max(0) |> min(length(boards) - 1)

          new_screen_state =
            Map.put(state.screen_state, :board_list, %{ss | selected_index: new_idx})

          {:update, %{state | screen_state: new_screen_state}, []}
        end
      end

      defp board_unread(%{unread_count: n}) when is_integer(n), do: n
      defp board_unread(_), do: 0

      defp domain_module(state, key) do
        get_in(state.session_context, [:domain, key]) || default_domain(key)
      end

      defp default_domain(:boards), do: Foglet.Boards
      defp default_domain(:threads), do: Foglet.Threads
      defp default_domain(:posts), do: Foglet.Posts
      defp default_domain(:markdown), do: Foglet.Markdown
    end
    ```

    ```elixir
    defmodule Foglet.TUI.Screens.ThreadList do
      @moduledoc """

      """

      import Raxol.View.Elements

      @spec render(map()) :: any()
      def render(state) do
        board = state.current_board
        threads = state.current_thread_list || []
        ss = get_in(state.screen_state, [:thread_list]) || %{selected_index: 0}

        sorted = sort_threads(threads)

        panel title: "Threads - #{board && board.name || "?"}", border: :single do
          StatusBar.render(%{
            handle: state.current_user && state.current_user.handle,
            location: "Threads: #{board && board.name || "?"}"
          })

          cond do
            sorted == [] and state.current_thread_list == [] ->
              text(content: "No threads in this board yet. Press [C] to compose.", color: :yellow)

            sorted == [] ->
              text(content: "Loading...", color: :bright_black)

            true ->
              box style: [flex_direction: :column, padding: 1] do
                for {thread, idx} <- Enum.with_index(sorted) do
                  marker = if idx == ss.selected_index, do: "▶ ", else: "  "

                  unread = Map.get(thread, :unread_count, 0)
                  unread_str = if unread > 0, do: " (#{unread})", else: ""

                  text(
                    content: "#{marker}#{sticky_mark}#{thread.title}#{unread_str}",
                    color: if(idx == ss.selected_index, do: :bright_green, else: :green)
                  )
                end
              end
          end

          KeyBar.render([
            {"↑/↓", "Select"},
            {"Enter", "Open"},
            {"C", "Compose"},
            {"Q", "Back"}
          ])
        end
      end

      @spec handle_key(map(), map()) :: {:update, map(), list()} | :no_match

      def handle_key(%{key: "down"}, state), do: move_selection(state, +1)
      def handle_key(%{key: "k"}, state), do: move_selection(state, -1)
    ```

    ```elixir
      @moduledoc """
      Post reader for a single thread (SSH-07, SSH-08, SSH-09).

      Renders posts via Foglet.Markdown.render/1. Paginates with n/p keys.
      Local read-pointer state (state.read_position) updates on each advance;
      flush to DB happens on screen transition via {:flush_read_pointers, _}
      command handled by TUI.App.update/2 (SSH-09).
      """

      alias Foglet.TUI.Widgets.{KeyBar, StatusBar}
      import Raxol.View.Elements

      @spec render(map()) :: any()
      def render(state) do
        thread = state.current_thread
        posts = state.posts || []
        ss = get_in(state.screen_state, [:post_reader]) || %{selected_post_index: 0}
        current_idx = ss.selected_post_index

        panel title: "Thread: #{thread && thread.title || "?"}", border: :single do
          StatusBar.render(%{
            handle: state.current_user && state.current_user.handle,
            location: "Reading: #{thread && thread.title || "?"}"
          })

          cond do
            posts == [] ->
              text(content: "Loading posts...", color: :bright_black)

            current_idx >= length(posts) ->
              text(content: "No more posts.", color: :yellow)

            true ->
              post = Enum.at(posts, current_idx)
              render_post(state, post, current_idx, length(posts))
          end

          KeyBar.render([
            {"N", "Next"},
            {"P", "Prev"},
            {"R", "Reply"},
            {"Q", "Back"}
          ])
        end
      end

      @spec handle_key(map(), map()) :: {:update, map(), list()} | :no_match
      def handle_key(%{key: key}, state) when key in ["n", "N", "space", "pagedown"] do
        advance_post(state, +1)
      end

      def handle_key(%{key: key}, state) when key in ["p", "P", "pageup"] do
        advance_post(state, -1)
      end

      def handle_key(%{key: key}, state) when key in ["r", "R"] do
        # Open composer as reply to currently-visible post
        posts = state.posts || []
        ss = get_in(state.screen_state, [:post_reader]) || %{selected_post_index: 0}
        reply_to = Enum.at(posts, ss.selected_post_index)

        new_state =
          state
          |> Map.put(:current_screen, :post_composer)
          |> Map.put(:composer_draft, "")
          |> put_in([:screen_state, :post_composer], %{
            mode: :edit,
            reply_to: reply_to,
            error: nil
          })

        {:update, new_state, []}
      end

      def handle_key(%{key: key}, state) when key in ["q", "Q"] do
        # SSH-09: flush read pointer on leave
        ctx = build_flush_context(state)

        new_state = %{
          state
          | current_screen: :thread_list,
            posts: nil,
            screen_state: Map.delete(state.screen_state, :post_reader)
        }

        {:update, new_state, [{:flush_read_pointers, ctx}]}
      end

      def handle_key(_key, _state), do: :no_match

      @doc "Called by App.update/2 on {:load_posts, thread_id}."
      @spec load_posts(map(), String.t()) :: {map(), list()}
      def load_posts(state, thread_id) do
        posts_mod = get_in(state.session_context, [:domain, :posts]) || Foglet.Posts

        if function_exported?(posts_mod, :list_posts, 1) do

      @doc """
      Flush board and thread read pointers to DB (SSH-09).
      Called by App.update/2 on {:flush_read_pointers, ctx}.
      """
      @spec flush_read_pointers(map(), map()) :: {map(), list()}
      def flush_read_pointers(state, ctx) do
        boards_mod = get_in(state.session_context, [:domain, :boards]) || Foglet.Boards
        threads_mod = get_in(state.session_context, [:domain, :threads]) || Foglet.Threads

        user_id = ctx[:user_id] || (state.current_user && state.current_user.id)

        # Best-effort: if the domain functions aren't defined (Phase 2 not executed),
        # skip silently. Real work happens once Phase 2 ships.
        if user_id do
          if ctx[:board_id] and function_exported?(boards_mod, :advance_board_read_pointer, 3) do
            boards_mod.advance_board_read_pointer(
              user_id,
              ctx[:board_id],
              ctx[:last_read_message_number] || 0
            )
          end

          if ctx[:thread_id] and function_exported?(threads_mod, :advance_read_pointer, 3) do
            threads_mod.advance_read_pointer(user_id, ctx[:thread_id], ctx[:last_read_post_id])
          end
        end

        # Clear local read_position for this thread now that it's persisted.
        new_rp =
          if ctx[:thread_id] do
            Map.delete(state.read_position, ctx[:thread_id])
          else
            state.read_position
          end

        {%{state | read_position: new_rp}, []}
      end

      defp advance_post(state, delta) do
        posts = state.posts || []

        if posts == [] do
          :no_match
        else
          ss = get_in(state.screen_state, [:post_reader]) || %{selected_post_index: 0}
          new_idx = (ss.selected_post_index + delta) |> max(0) |> min(length(posts) - 1)
          post = Enum.at(posts, new_idx)

          # Local read-pointer advance — flushed on screen transition
          new_rp =
            if post and state.current_thread do
              Map.put(state.read_position, state.current_thread.id, %{
                last_read_post_id: post.id,
                last_read_message_number: Map.get(post, :message_number, 0)
              })
            else
              state.read_position
            end

          new_screen_state =
            Map.put(state.screen_state, :post_reader, %{
              ss
              | selected_post_index: new_idx
            })

          {:update,
           %{state | screen_state: new_screen_state, read_position: new_rp}, []}
        end
      end

      defp render_post(state, post, idx, total) do
        markdown_mod = get_in(state.session_context, [:domain, :markdown]) || Foglet.Markdown

        rendered =
          if function_exported?(markdown_mod, :render, 1) do
            markdown_mod.render(post.body || "")
          else
            post.body || ""
          end

        box style: [flex_direction: :column, padding: 1] do
          text(content: "Post #{idx + 1} of #{total}", color: :bright_black)
          text(content: "")

          author = get_post_author(post)

          text(content: "By @#{author} at #{post.inserted_at}", color: :bright_black)
          text(content: "")
          text(content: rendered, color: :green)
        end
      end

      defp get_post_author(%{user: %{handle: h}}), do: h
      defp get_post_author(_), do: "unknown"

      defp build_flush_context(state) do
        thread = state.current_thread
        board = state.current_board
        pos = state.read_position[(thread && thread.id) || nil] || %{}

        %{
          user_id: state.current_user && state.current_user.id,
          board_id: board && board.id,
          thread_id: thread && thread.id,
          last_read_post_id: pos[:last_read_post_id],
          last_read_message_number: pos[:last_read_message_number]
        }
      end
    ```

    ```elixir
          box style: [flex_direction: :column, padding: 1] do
            if ss.reply_to do
              text(content: "Replying to @#{get_handle(ss.reply_to)}:", color: :bright_black)
              text(content: quote_preview(ss.reply_to), color: :bright_black)
              text(content: "")
            end

            case ss.mode do
              :edit -> render_editor(draft)
              :preview -> render_preview(state, draft)
            end

            if ss.error do
              text(content: "")
              text(content: ss.error, color: :red)
            end

            text(content: "")
            text(content: "#{String.length(draft)} / #{max_len(state)} chars", color: :bright_black)
          end

          KeyBar.render([
            {"Tab", if(ss.mode == :edit, do: "Preview", else: "Edit")},
            {"Ctrl+S", "Send"},
            {"Ctrl+C", "Cancel"}
          ])
        end
      end

      @spec handle_key(map(), map()) :: {:update, map(), list()} | :no_match
      def handle_key(%{key: "tab"}, state), do: toggle_mode(state)

      def handle_key(%{key: "ctrl_s"}, state), do: submit(state)
      def handle_key(%{key: "ctrl_c"}, state), do: cancel(state)

      def handle_key(%{key: "backspace"}, state) do
        draft = state.composer_draft || ""
        new_draft = String.slice(draft, 0, max(String.length(draft) - 1, 0))
        {:update, %{state | composer_draft: new_draft}, []}
      end

      def handle_key(%{key: key}, state) when is_binary(key) and byte_size(key) == 1 do
        draft = state.composer_draft || ""
        new_draft = draft <> key

        # Optionally surface a warning once the limit is reached, but still allow
        # typing; enforcement is final on Ctrl+S.
        {:update, %{state | composer_draft: new_draft}, []}
      end

      def handle_key(%{key: "enter"}, state) do
        draft = state.composer_draft || ""
        {:update, %{state | composer_draft: draft <> "\n"}, []}
      end

      def handle_key(_key, _state), do: :no_match

      # --- Private ---

      defp title_for(nil, nil), do: "New Thread"
      defp title_for(nil, thread), do: "Reply to: #{thread.title}"
      defp title_for(reply_to, _), do: "Reply to post ##{Map.get(reply_to, :message_number, "?")}"

      defp render_editor(draft) do
        # Using plain text rendering of the buffer. If Raxol's multi_line_input
        # widget is available, swap in: multi_line_input(value: draft, ...)
        # For unit-testability and to avoid Raxol-version-specific widget issues,
        # render the buffer as text here and use the same display contract.
        text(content: draft, color: :green)
      end

      defp render_preview(state, draft) do
        markdown_mod = get_in(state.session_context, [:domain, :markdown]) || Foglet.Markdown

        rendered =
          if function_exported?(markdown_mod, :render, 1) do
            markdown_mod.render(draft)
          else
            draft
          end

        text(content: rendered, color: :green)
      end

      defp quote_preview(post) do
        body = Map.get(post, :body, "")
        body
        |> String.split("\n")
        |> Enum.take(5)
        |> Enum.map(fn line -> "> #{line}" end)
        |> Enum.join("\n")
      end

      defp get_handle(%{user: %{handle: h}}), do: h
      defp get_handle(_), do: "unknown"

      defp toggle_mode(state) do
        ss =
          get_in(state.screen_state, [:post_composer]) ||
            %{mode: :edit, reply_to: nil, error: nil}

        new_mode = if ss.mode == :edit, do: :preview, else: :edit
        new_ss = Map.put(ss, :mode, new_mode)

        {:update,
         %{state | screen_state: Map.put(state.screen_state, :post_composer, new_ss)}, []}
      end

      defp submit(state) do
        draft = state.composer_draft || ""
        ss = get_in(state.screen_state, [:post_composer]) || %{reply_to: nil}
        max = max_len(state)

        cond do
          String.trim(draft) == "" ->
            modal = %{type: :error, message: "Post body cannot be empty."}
            {:update, %{state | modal: modal}, []}

          String.length(draft) > max ->
            modal = %{
              type: :error,
              message: "Post body exceeds maximum length of #{max} characters (D-31)."
            }

            {:update, %{state | modal: modal}, []}

          true ->
            do_submit(state, ss, draft)
        end
      end

      defp do_submit(state, ss, draft) do
        posts_mod = get_in(state.session_context, [:domain, :posts]) || Foglet.Posts

        if function_exported?(posts_mod, :create_reply, 3) and state.current_thread do
          attrs = %{body: draft}
          reply_to_id = ss[:reply_to] && ss[:reply_to].id
          attrs = if reply_to_id, do: Map.put(attrs, :reply_to_id, reply_to_id), else: attrs

          case posts_mod.create_reply(state.current_thread, state.current_user, attrs) do
            {:ok, _post} ->
              {:update,
               %{
                 state
                 | current_screen: :post_reader,
                   composer_draft: nil,
                   screen_state: Map.delete(state.screen_state, :post_composer)
               }, [{:load_posts, state.current_thread.id}]}

            {:error, _cs} ->
              modal = %{type: :error, message: "Failed to create post."}
              {:update, %{state | modal: modal}, []}
          end
        else
          # Submit without a live Posts module — treat as dev-mode stub
          modal = %{type: :info, message: "Submitted (dev-mode; Phase 2 not fully wired)."}
          {:update,
           %{
             state
             | modal: modal,
               current_screen: :thread_list,
               composer_draft: nil,
               screen_state: Map.delete(state.screen_state, :post_composer)
           }, []}
        end
      end

      defp cancel(state) do
        # D-30: immediate, no confirmation
        new_state = %{
          state
          | current_screen: :thread_list,
            composer_draft: nil,
            screen_state: Map.delete(state.screen_state, :post_composer)
        }

        {:update, new_state, []}
      end

      defp max_len(state) do
        case get_in(state.session_context, [:max_post_length]) do
          n when is_integer(n) and n > 0 -> n
          _ -> safe_config_get("max_post_length", @default_max_post_length)
        end
      end

      defp safe_config_get(key, default) do
        try do
          case Config.get!(key) do
            n when is_integer(n) -> n
            _ -> default
          end
        rescue
          _ -> default
        end
      end
    end
    ```

## Trust Boundaries

| Boundary | Description |

| User keystrokes → composer draft | Untrusted multi-line input stored in state.composer_draft; length enforced at submit. |

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-3-04-01 | Tampering | Post body exceeding max_post_length | mitigate | PostComposer.submit/1 checks `String.length(draft) > max_len(state)` and blocks submission with error modal (D-31); enforcement happens BEFORE Foglet.Posts.create_reply/3 call |

| T-3-04-03 | DoS | Giant composer draft in memory | accept | Draft is bounded at max_post_length (8192 default) per Session. Absent DoS vector — each Session has one draft slot. |
| T-3-04-04 | Spoofing | Reply to a post in a different thread | mitigate | Composer stores reply_to from PostReader's current selection; create_reply/3 validates at the domain layer (Phase 2) |
| T-3-04-05 | Information Disclosure | Showing unauthorized board unread counts | accept | Foglet.Boards.list_subscribed_boards/1 is the authoritative visibility filter (Phase 2); TUI renders what the context returns. |
| T-3-04-06 | Tampering | Skipping read-pointer flush by fast-disconnect | accept | On abrupt disconnect, flush doesn't run — worst case, user re-reads a few posts. Acceptable per D-21 (no E2E session tests in Phase 3). |
| T-3-04-07 | Elevation of Privilege | Unauthenticated user reaching PostComposer | mitigate | App.init/2 routes unauthenticated sessions to :login; `{:navigate, :post_composer}` is only emitted from authenticated screens (MainMenu/ThreadList/PostReader) — all of which require a `current_user` set by Plan 03's Login/Verify success paths |

- Tab toggle between edit/preview is purely a view concern; no state mutation of the draft happens on Tab. Safe.
