def render_with_metadata(title, metadata, selected, unread?, theme, opts \\ []) do
  width = Keyword.get(opts, :width, 80)
  marker = if selected, do: "> ", else: "  "

  {title_part, padding_part, metadata_part} = compute_parts(marker, title, metadata, width)
  {title_style, metadata_style, padding_style} = styles_for(selected, unread?, theme)

  row style: %{gap: 0} do
    [
      text(title_part, title_style),
      text(padding_part, padding_style),
      text(metadata_part, metadata_style)
    ]
  end
end

defp compute_parts(marker, title, metadata, width) do
  marker_width = TextWidth.display_width(marker)
  metadata_width = TextWidth.display_width(metadata)
  min_gap = 2

  max_title_body = max(width - marker_width - min_gap - metadata_width, 0)
  title_body = truncate_title(title, max_title_body)

  title_part = marker <> title_body
  title_part_width = marker_width + TextWidth.display_width(title_body)

  padding_width =
    (width - title_part_width - metadata_width) |> max(0) |> min(width)

  padding_part = TextWidth.pad_trailing("", padding_width)

  {title_part, padding_part, metadata}
end

defmodule Foglet.TUI.Widgets.List.RichRow do
  @moduledoc """
  Rich row renderer for Foglet BBS selection lists (RICHROW-01, THREADS-01,
  THREADS-02).

  A stateless rendering widget â€” no internal state, no `init/1`, no
  `handle_event/2`. Callers (screens) own all data and selection index.

  ## Public API

      RichRow.render(
        title: "Thread title",
        metadata: "@alice آ· 3 posts آ· 2h ago",
        state_cluster: [:unread, :sticky],
        selected: true,
        theme: theme,
        width: 80,         # optional, default 80
        focus_marker: "â–Œ ",  # optional, default "â–Œ "
        emphasis: :bold      # optional, e.g. :bold for unread
      )

  ## Required keyword keys

    * `:title`         â€” String, the row's primary text.
    * `:metadata`      â€” String, right-aligned trailer (e.g. `"@handle آ· N posts آ· age"`).
    * `:state_cluster` â€” list of state atoms (e.g. `[]`, `[:unread]`,
                         `[:sticky, :locked]`). Determines glyphs in the
                         leading cluster.
    * `:selected`      â€” boolean, true if this row is the currently focused row.
    * `:theme`         â€” `%Foglet.TUI.Theme{}` for slot lookups.

  ## Optional keyword keys

    * `:width`         â€” non-neg integer, total content width in display cells
                         (default 80, matching `ListRow.render_with_metadata/6`).
    * `:focus_marker`  â€” String, leading marker when `selected: true`
                         (default `"â–Œ "`).
    * `:emphasis`      â€” atom, applied to the title text (e.g. `:bold` for
                         unread). Currently only `:bold` is recognized.

  ## Supported state atoms (Phase 20)

    * `:unread`  â€” `â—†` (U+25C6) in `theme.accent.fg` + `:bold`
    * `:sticky`  â€” `â—ڈ` (U+25CF) in `theme.info.fg`
    * `:locked`  â€” `âڑ؟` (U+26BF) in `theme.warning.fg`

  Read state (no `:unread` atom in the list) renders as 1-cell whitespace at
  the unread position. The cluster's total display width is **fixed** at
  `@cluster_width` regardless of which atoms are present, so columns stay
  aligned across all rows.

  ## Reserved for Phase 21+

    * `:subscribed`, `:required`, `:category` â€” currently render as whitespace.
      Phase 21 (BoardList) and Phase 25 (operator surfaces) will add glyph
      mappings without changing this public API.

  ## Size-contract priority

  At any terminal width:

    1. The leading state cluster always renders fully (`@cluster_width` cells).
    2. The metadata always renders fully.
    3. The title truncates first with `"â€¦"` when needed.

  The 20-cell minimum title attempt established by
  `ListRow.render_with_metadata/6` is preserved â€” below that the row falls
  back to a compact title + ellipsis treatment but the cluster and metadata
  still render in full.

  ## Theme routing

  No hardcoded color atoms appear in this module. Every styling decision
  goes through `theme.slot.fg`, `theme.slot.bg`, or `theme.slot.style`.

  Honours decisions: D-01 (keyword API), D-02 (state atoms), D-03 (fixed
  cluster width via `TextWidth.display_width/1`), D-04 (this moduledoc),
  D-05 (glyph mapping), D-06 (theme-slot routing), D-07 (single Unicode
  set, no ASCII fallback), D-08 (`â–Œ` focus marker), D-09 (selected/unselected
  treatment), D-10 (selection âٹ¥ state-cluster).
  """

  import Raxol.Core.Renderer.View

  alias Foglet.TUI.TextWidth
  alias Foglet.TUI.Theme

  # Cluster format: "USL " â€” 3 glyph slots + 1 trailing separator space.
  # Each glyph (or its replacement space) is exactly 1 display cell.
  @cluster_width 4

  @focus_marker "â–Œ "
  @no_focus_marker "  "
  @ellipsis "â€¦"
  @min_title_length 20

  @glyph_unread "â—†"
  @glyph_sticky "â—ڈ"
  @glyph_locked "âڑ؟"
  @glyph_space " "

  @doc """
  Renders a rich row from a keyword list. See moduledoc for the contract.
  """
  @spec render(keyword()) :: any()
  def render(opts) when is_list(opts) do
    title = Keyword.fetch!(opts, :title)
    metadata = Keyword.fetch!(opts, :metadata)
    state_cluster = Keyword.fetch!(opts, :state_cluster)
    selected = Keyword.fetch!(opts, :selected)
    %Theme{} = theme = Keyword.fetch!(opts, :theme)

    width = Keyword.get(opts, :width, 80)
    focus_marker = Keyword.get(opts, :focus_marker, @focus_marker)
    emphasis = Keyword.get(opts, :emphasis, nil)

    marker = if selected, do: focus_marker, else: @no_focus_marker
    {cluster_str, cluster_styles} = render_cluster_string(state_cluster, theme)

    {left_part, padding_part, metadata_part} =
      compute_parts(marker, cluster_str, title, metadata, width)

    {title_style, metadata_style, padding_style} =
      styles_for(selected, emphasis, theme)

    # When selected, the entire row uses theme.selected.{fg,bg} + :bold so
    # the focus marker and cluster glyphs adopt the selection styling and
    # the unique :bg property satisfies SPEC requirement 4 acceptance.
    cluster_elements =
      if selected do
        # Single text node â€” selection styling is uniform across the cluster.
        [text(cluster_str, title_style)]
      else
        # One text node per glyph slot so each glyph carries its own theme
        # slot fg. The trailing separator space adopts the title style.
        cluster_styles
        |> Enum.with_index()
        |> Enum.map(fn {style, idx} ->
          char = String.slice(cluster_str, idx..idx)
          text(char, style)
        end)
      end

    # Compose: marker text, then cluster (one or more text nodes), then
    # title+padding+metadata. Use a row with gap: 0 so cell positions align.
    title_only = String.replace_prefix(left_part, marker <> cluster_str, "")

    row style: %{gap: 0} do
      [
        text(marker, title_style),
        cluster_elements,
        text(title_only, title_style),
        text(padding_part, padding_style),
        text(metadata_part, metadata_style)
      ]
      |> List.flatten()
    end
  end

  # ---------------------------------------------------------------
  # Private â€” cluster rendering
  # ---------------------------------------------------------------

  # Returns {cluster_string, [style_kw, style_kw, style_kw, style_kw]} where
  # the four style keyword lists correspond to positions:
  #   0: unread slot   1: sticky slot   2: locked slot   3: trailing separator
  #
  # cluster_string display width is exactly @cluster_width regardless of
  # which atoms are present.
  @spec render_cluster_string([atom()], Theme.t()) :: {String.t(), [keyword()]}
  defp render_cluster_string(state_atoms, theme) when is_list(state_atoms) do
    {u_char, u_style} =
      if :unread in state_atoms,
        do: {@glyph_unread, [fg: theme.accent.fg, style: [:bold]]},
        else: {@glyph_space, [fg: theme.dim.fg]}

    {s_char, s_style} =
      if :sticky in state_atoms,
        do: {@glyph_sticky, [fg: theme.info.fg]},
        else: {@glyph_space, [fg: theme.dim.fg]}

    {l_char, l_style} =
      if :locked in state_atoms,
        do: {@glyph_locked, [fg: theme.warning.fg]},
        else: {@glyph_space, [fg: theme.dim.fg]}

    cluster = u_char <> s_char <> l_char <> @glyph_space

    # Verify @cluster_width invariant at runtime: the rendered cluster string
    # MUST measure exactly @cluster_width display cells. If this ever fails,
    # a glyph in the map has a non-1 display width and the cluster contract
    # is broken. Pitfall 1 / Pitfall 3 from RESEARCH.md.
    ^cluster_width = TextWidth.display_width(cluster)

    {cluster, [u_style, s_style, l_style, [fg: theme.dim.fg]]}
  end

  # Compile-time helper to allow the runtime guard above.
  @cluster_width_local @cluster_width
  defp cluster_width, do: @cluster_width_local

  # ---------------------------------------------------------------
  # Private â€” width math
  # ---------------------------------------------------------------

  @spec compute_parts(String.t(), String.t(), String.t(), String.t(), pos_integer()) ::
          {String.t(), String.t(), String.t()}
  defp compute_parts(marker, cluster_str, title, metadata, width) do
    marker_width = TextWidth.display_width(marker)
    cluster_width = TextWidth.display_width(cluster_str)
    metadata_width = TextWidth.display_width(metadata)
    min_gap = 2

    max_title_body =
      max(width - marker_width - cluster_width - min_gap - metadata_width, 0)

    title_body = truncate_title(title, max_title_body)

    left_part = marker <> cluster_str <> title_body
    left_width = marker_width + cluster_width + TextWidth.display_width(title_body)

    padding_width =
      (width - left_width - metadata_width) |> max(0) |> min(width)

    padding_part = TextWidth.pad_trailing("", padding_width)

    {left_part, padding_part, metadata}
  end

  @spec truncate_title(String.t(), non_neg_integer()) :: String.t()
  defp truncate_title(title, max_width) do
    title_width = TextWidth.display_width(title)

    cond do
      title_width <= max_width ->
        title

      max_width >= @min_title_length ->
        TextWidth.truncate(title, max_width)

      max_width >= 2 ->
        TextWidth.truncate(title, min(max_width, TextWidth.display_width(@ellipsis) + 1))

      true ->
        TextWidth.truncate(title, max_width)
    end
  end

  # ---------------------------------------------------------------
  # Private â€” style dispatch
  # ---------------------------------------------------------------

  # Selected rows: title, metadata, and padding all carry the selection
  # styling (fg, bg, :bold). The :bg slot is the uniqueness property
  # SPEC requirement 4 acceptance relies on (every shipped theme defines
  # selected.bg in theme.ex:113, 129, 145, 161, 177, 193, 209, 225, 241).
  defp styles_for(true, _emphasis, theme) do
    selected_style = Map.get(theme.selected, :style, [:bold])

    sel_kw = [
      fg: theme.selected.fg,
      bg: theme.selected.bg,
      style: selected_style
    ]

    {sel_kw, sel_kw, sel_kw}
  end

  # Unselected + emphasis :bold (unread, not selected): title in accent.fg + :bold.
  defp styles_for(false, :bold, theme) do
    {
      [fg: theme.accent.fg, style: [:bold]],
      [fg: theme.dim.fg],
      [fg: theme.dim.fg]
    }
  end

  # Unselected + no emphasis: plain unselected styling.
  defp styles_for(false, _emphasis, theme) do
    {
      [fg: theme.unselected.fg],
      [fg: theme.dim.fg],
      [fg: theme.dim.fg]
    }
  end
end
