```elixir
# ---------------------------------------------------------------------------
# Phase 20 — ThreadList rich-row size contracts (RICHROW-01 / THREADS-01)
# ---------------------------------------------------------------------------

describe "thread_list — size contract" do
  setup do
    now = DateTime.utc_now()

    long_title = String.duplicate("x", 100)

    threads = [
      %Foglet.Threads.ThreadEntry{
        id: "t1",
        title: "Sticky " <> long_title,
        board_id: "b1",
        sticky: true,
        locked: false,
        has_unread: true,
        post_count: 5,
        last_post_at: now,
        created_by: %{handle: "alice"}
      },
      %Foglet.Threads.ThreadEntry{
        id: "t2",
        title: "Locked thread title",
        board_id: "b1",
        sticky: false,
        locked: true,
        has_unread: false,
        post_count: 3,
        last_post_at: DateTime.add(now, -3600, :second),
        created_by: %{handle: "alice"}
      },
      %Foglet.Threads.ThreadEntry{
        id: "t3",
        title: "Plain thread",
        board_id: "b1",
        sticky: false,
        locked: false,
        has_unread: false,
        post_count: 1,
        last_post_at: DateTime.add(now, -7200, :second),
        created_by: %{handle: "alice"}
      }
    ]

    user = %Foglet.Accounts.User{id: "u1", handle: "alice"}
    %{threads: threads, user: user}
  end

  for {width, height} <- [{64, 22}, {80, 24}, {132, 50}] do
    @tag width: unquote(width), height: unquote(height)
    test "at #{unquote(width)}x#{unquote(height)} cluster, metadata visible; title truncates only when forced",
         %{threads: threads, user: user} do
      width = unquote(width)
      height = unquote(height)

      state =
        %Foglet.TUI.App{
          current_screen: :thread_list,
          current_user: user,
          current_board: %{id: "b1", name: "General", slug: "general"},
          session_context: %{},
          terminal_size: {width, height},
          current_thread_list: threads,
          screen_state: %{thread_list: %{selected_index: 0}}
        }
        |> Map.from_struct()

      tree = ThreadList.render(state)
      positioned = apply_at_size(tree, {width, height})
      flat = flatten_text(positioned)

      # (a) Cluster glyphs render fully — sticky+unread on t1, locked on t2
      assert flat =~ "◆", "expected unread glyph at #{width}x#{height}, got: #{inspect(flat)}"
      assert flat =~ "●", "expected sticky glyph at #{width}x#{height}, got: #{inspect(flat)}"
      assert flat =~ "⚿", "expected locked glyph at #{width}x#{height}, got: #{inspect(flat)}"

      # (b) Metadata renders fully
      assert flat =~ "@alice"
      assert flat =~ "·"

      # (c) Title truncates with … only when width forces it. The 100-char
      # sticky title cannot fit at 64 cells; it MAY also not fit at 80 cells
      # (depending on cluster + metadata budget). At 132 cells it should fit.
      if width <= 80 do
        assert flat =~ "…",
               "expected title truncation at width #{width}, got: #{inspect(flat)}"
      end

      # (d) No positioned text element exceeds the width budget
      for element <- text_elements(positioned) do
        text = Map.fetch!(element, :text)

        assert element.x >= 0
        assert element.y >= 0
        assert element.x + TextWidth.display_width(text) <= width,
               "element #{inspect(text)} at x=#{element.x} exceeds width #{width}"
      end

      # (e) The legacy [S] prefix is gone at every size
      refute flat =~ "[S] "
    end
  end
end
```
