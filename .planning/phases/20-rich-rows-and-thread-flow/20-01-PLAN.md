The RichRow public API the tests will exercise (per D-01 / D-02):

```elixir
# Required keyword args
RichRow.render(
  title: String.t(),
  metadata: String.t(),
  state_cluster: [atom()],   # e.g. [], [:unread], [:sticky, :locked]
  selected: boolean(),
  theme: %Foglet.TUI.Theme{}
)

# Optional keyword args
#   :width         (default 80)
#   :focus_marker  (default "▌ ")
#   :emphasis      (e.g. :bold for unread)
```

```elixir
flatten_text(tree) :: String.t()
assert_text_run(tree, content :: String.t(), expected_keys :: keyword()) :: map()
color_atom_leaked?(serialized :: String.t(), color :: String.t()) :: boolean()
color_names() :: ["red", "green", "cyan", "yellow", "blue", "magenta", "white", "black"]
```

Theme construction: `theme = Foglet.TUI.Theme.default()` returns a `%Theme{}`
with `accent`, `info`, `badge`, `warning`, `dim`, `selected`, `unselected` slots.

```elixir
defmodule Foglet.TUI.Widgets.List.RichRowTest do
  use ExUnit.Case, async: true

  import Foglet.TUI.WidgetHelpers,
    only: [flatten_text: 1, assert_text_run: 3, color_atom_leaked?: 2, color_names: 0]

  alias Foglet.TUI.TextWidth
  alias Foglet.TUI.Theme
  alias Foglet.TUI.Widgets.List.RichRow

  setup do
    %{theme: Theme.default()}
  end

  describe "render/1 — selection × state × metadata matrix (RICHROW-01)" do
    test "unselected, no state, with metadata renders left-padded title and right-aligned metadata", %{theme: theme}
    test "unselected, [:unread], with metadata applies emphasis :bold and accent fg to title", %{theme: theme}
    test "unselected, [:sticky], with metadata routes sticky glyph through theme.info.fg", %{theme: theme}
    test "unselected, [:locked], with metadata routes locked glyph through theme.warning.fg", %{theme: theme}
    test "selected, no state, with metadata renders ▌ marker and theme.selected.{fg,bg} + :bold", %{theme: theme}
    test "selected, [:unread], with metadata stacks selection styling over unread emphasis", %{theme: theme}
    test "selected, [:sticky, :locked], with metadata renders both glyphs in cluster", %{theme: theme}
    test "unselected, no state, no metadata renders title without metadata text", %{theme: theme}
  end

  describe "render/1 — focused-row uniqueness (THREADS-02 / SPEC req 4)" do
    test "focused unread row has at least one styling property unselected rows do not share", %{theme: theme}
    test "focused read row has at least one styling property unselected rows do not share", %{theme: theme}
  end

  describe "render/1 — cluster fixed-width invariant (D-03 / SPEC req 2 acceptance d)" do
    test "leading cluster display width is identical across [], [:unread], [:sticky], [:locked], and [:unread, :sticky, :locked]", %{theme: theme}
  end

  describe "render/1 — generic state-atom acceptance (RICHROW-01 / SPEC req 6)" do
    test "renders [:subscribed, :required] cluster as visual whitespace without referencing Foglet.Threads", %{theme: theme}
  end

  describe "render/1 — 64-cell width contract (SPEC req 5)" do
    test "long title at width: 64 truncates with … while cluster and metadata render fully", %{theme: theme}
    test "below 20-cell budget, title still ends with … and full metadata is preserved", %{theme: theme}
  end

  describe "render/1 — theme-routing hygiene (RICHROW-01)" do
    test "no hardcoded color atom leaks for any selection × state × metadata combination", %{theme: theme}
  end

  describe "render/1 — focus marker convention (D-08)" do
    test "selected row begins with ▌ marker, unselected row begins with two-space gutter", %{theme: theme}
  end

  describe "render/1 — glyph identity (D-05)" do
    test "[:unread] cluster contains ◆ (U+25C6)", %{theme: theme}
    test "[:sticky] cluster contains ● (U+25CF)", %{theme: theme}
    test "[:locked] cluster contains ⚿ (U+26BF)", %{theme: theme}
  end
end
```

For each test body:

1. **Matrix tests (1-8)**: Build a render call like

   ```elixir
   tree =
     RichRow.render(
       title: "Thread alpha",
       metadata: "@alice · 3 posts · 2h ago",
       state_cluster: [:unread],
       selected: false,
       theme: theme,
       width: 80,
       emphasis: :bold
     )

   flat = flatten_text(tree)
   assert flat =~ "Thread alpha"
   assert flat =~ "@alice · 3 posts · 2h ago"
   assert flat =~ "◆"
   assert_text_run(tree, "Thread alpha", fg: theme.accent.fg, style: [:bold])
   ```

   Use `assert_text_run/3` for style-property assertions. Use `flatten_text/1` + `=~`
   for content presence. NEVER assert via `inspect(tree) =~ "..."` for style.

2. **Focused-row uniqueness (Tests 9-10)**: Render two rows in one call sequence —
   one with `selected: true` and one with `selected: false`, both with the same
   `state_cluster` — then walk both `text_runs/1` results and assert there exists
   at least one style key (`:bg`, or `:style` containing `:bold`) where the focused
   row's value differs from every non-focused row's value. Acceptable shape:

   ```elixir
   focused = RichRow.render(title: "F", metadata: "m", state_cluster: [:unread], selected: true, theme: theme, emphasis: :bold)
   normal  = RichRow.render(title: "N", metadata: "m", state_cluster: [:unread], selected: false, theme: theme, emphasis: :bold)

   focused_runs = Foglet.TUI.WidgetHelpers.text_runs(focused)
   normal_runs  = Foglet.TUI.WidgetHelpers.text_runs(normal)

   focused_bgs = focused_runs |> Enum.map(&Map.get(&1, :bg))
   normal_bgs  = normal_runs  |> Enum.map(&Map.get(&1, :bg))

   assert Enum.any?(focused_bgs, fn bg -> bg != nil and bg not in normal_bgs end),
          "expected focused row to have a :bg value no non-focused run shares"
   ```

3. **Cluster fixed-width invariant (Test 11)**: For each `state_cluster` in
   `[[], [:unread], [:sticky], [:locked], [:unread, :sticky, :locked]]`, render
   with `width: 80` and assert `TextWidth.display_width(flatten_text(tree))` is
   identical across all five renders (subject to the same title and metadata).
   This locks D-03's "fixed cluster width via TextWidth" contract.

4. **Generic state atoms (Test 12)**: Render with `state_cluster: [:subscribed, :required]`
   and assert no `KeyError`/`FunctionClauseError`. Assert the cluster pads to
   the same display width as `state_cluster: []`. Do NOT alias or reference
   `Foglet.Threads` anywhere in this test.

5. **64-cell budget (Test 13)**: Build a 100-character title (`String.duplicate("x", 100)`),
   render with `width: 64`, assert:
   - `flat =~ "◆"` (cluster fully rendered)
   - `flat =~ "@alice"` (metadata fully rendered)
   - `flat =~ "…"` (title truncated)
   - `TextWidth.display_width(flat) <= 64`

6. **Theme-routing hygiene (Test 15)**: After rendering each combination, run

   ```elixir
   serialized = inspect(tree, limit: :infinity)
   for color <- color_names() do
     refute color_atom_leaked?(serialized, color),
            "found hardcoded :#{color} atom for combo #{label}"
   end
   ```

7. **Focus marker (Test 16)**: Assert `flatten_text(selected_tree)` starts with
   `"▌ "` (U+258C + space) and `flatten_text(unselected_tree)` starts with
   `"  "` (two spaces).

8. **Glyph identity (Tests 17-19)**: Use direct `flatten_text(tree) =~ "◆"`
   for `[:unread]`, `=~ "●"` for `[:sticky]`, `=~ "⚿"` for `[:locked]`.
