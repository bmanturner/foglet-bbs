Viewport API (from vendor/raxol/lib/raxol/ui/components/display/viewport.ex, verified via direct source inspection in 07-RESEARCH.md):

```elixir
alias Raxol.UI.Components.Display.Viewport

# init/1 returns {:ok, state_map}:
{:ok, vp} = Viewport.init(%{
  id: "post_reader_vp",
  children: [],                # list of Raxol view elements — one per scrollable line
  visible_height: 10,          # number of children rendered at once; updated at runtime
  scroll_top: 0,               # starting offset
  show_scrollbar: false,       # BBS aesthetic — omit scrollbar chrome
  focused: false               # no focus manager wired in PostReader
})

# vp state shape (map):
#   %{id, children, content_source, scroll_top, scroll_left,
#     visible_height, visible_width, content_height, style, theme,
#     show_scrollbar, focused}
# We only read/modify scroll_top and update children/visible_height via messages.

# update/2 returns {new_state, []}:
{vp, []} = Viewport.update({:set_children, lines}, vp)           # replaces children AND re-clamps scroll_top
{vp, []} = Viewport.update({:set_visible_height, h}, vp)         # updates visible_height AND re-clamps scroll_top
{vp, []} = Viewport.update({:scroll_by, +1}, vp)                 # j — relative scroll, auto-clamped
{vp, []} = Viewport.update({:scroll_by, -1}, vp)                 # k
{vp, []} = Viewport.update({:scroll_to, 0}, vp)                  # reset to top (on N/P/space)

# render/2 returns %{type: :row, id: _, style: _, children: [col, maybe_scrollbar]}:
rendered = Viewport.render(vp, %{})
# The map is a valid Raxol view element — can be used as a child of column/row/box.
```

IMPORTANT: Viewport's `clamp_scroll/3` clamps `scroll_top` to `[0, max(0, content_height - visible_height)]` automatically on every update. We do NOT need our own max_offset math.

defp scroll_post(state, delta) do
  posts = state.posts || []

  if posts == [] do
    :no_match
  else
    ss = get_screen_state(state)
    post = Enum.at(posts, ss.selected_post_index)

    if is_nil(post) do
      :no_match
    else
      {w, _h} = state.terminal_size || {80, 24}
      {new_vp, []} = Viewport.update({:scroll_by, delta}, ss.viewport)
      ss = %{ss | viewport: new_vp}
      ss = warm_cache(ss, state, post, w)
      new_screen_state = Map.put(state.screen_state, :post_reader, ss)
      {:update, %{state | screen_state: new_screen_state}, []}
    end
  end
end

Note: scroll_post no longer needs `PostCard.body_line_count/1` (Viewport clamps via content_height which is derived from children list length — and we haven't set children yet at key-press time).
