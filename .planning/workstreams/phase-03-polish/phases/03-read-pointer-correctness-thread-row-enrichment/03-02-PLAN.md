# Plan 02 — Threads.list_threads/2: Unread Annotation (has_unread virtual field)

<objective>
Add a sibling function `Foglet.Threads.list_threads/2` that returns threads annotated with a virtual `:has_unread` boolean per the current user's thread read pointers. This is the data backbone for LIST-03 D-04 (bold-on-unread thread titles). Existing `list_threads/1` is NOT changed — it stays as the "no user context" variant used by tests and admin paths, preserving backwards compatibility with all current callers including `test/foglet_bbs/tui/screens/thread_list_test.exs` and `app.ex:390` (`threads_mod.list_threads(board_id)`).

**Strategy choice (resolves CONTEXT.md "Claude's Discretion" Thread unread detection):** Approach A+B hybrid. Add a **new 2-arity function** (`list_threads/2` that takes a `user_id`) — this is "Approach A: add user_id param" without breaking "Approach B: preserve 1-arity". The function issues a single query that LEFT JOINs thread read pointers for the caller's user and compares `thread.last_post_at > trp.last_read_at` to compute a per-row `has_unread` boolean. This avoids Approach C's pure-Elixir merge (slower on large boards) and the Approach A purist form's backwards-incompatibility cost.

**Why `last_post_at > trp.last_read_at` (timestamp comparison) and not `thread.post_count > trp.<count>`:** The `thread_read_pointers` table stores `last_read_at :: utc_datetime_usec` and `last_read_post_id :: uuid` — it does NOT store a post count or message number. Timestamp comparison is the cheapest, most faithful "any activity since I last looked" signal available from the existing schema. Per D-04, the UI only needs a boolean ("any unread"), not a count — so this signal is sufficient.

**Virtual field placement:** `:has_unread` is NOT added to the `Foglet.Threads.Thread` Ecto schema as a `field ... virtual: true` — that would pollute the schema with a field only meaningful in list contexts. Instead, the new function returns plain maps merged from each thread struct with `has_unread` added (`Map.put(struct_as_map, :has_unread, bool)`). Callers (ThreadList screen) read `Map.get(thread, :has_unread, false)` — the same pattern already used for `:unread_count` on Board structs at `board_list.ex:108`. This keeps Thread struct pristine and matches the workstream's existing "annotate outside the schema" convention.

**New-user / no-read-pointer semantics:** If the user has no `thread_read_pointers` row for a given thread, the LEFT JOIN yields `trp.last_read_at IS NULL`. A thread the user has never opened must be treated as unread (any post within it is unseen). The coalesce-and-compare handles this: `t.last_post_at > COALESCE(trp.last_read_at, '1970-01-01 00:00:00 UTC')` — i.e., NULL read-pointer collapses to the Unix epoch, making any `last_post_at > epoch` true. An empty thread (`last_post_at IS NULL`) is treated as read (no activity to be unread about).
</objective>

```elixir
  @doc """
  List all non-deleted threads in a board, annotated with `:has_unread` for
  the given user (LIST-03).

  Returns a list of maps (not `Thread.t()` structs) — each map merges the
  thread's fields with a boolean `:has_unread` virtual field computed
  from `thread_read_pointers`. A thread is `has_unread: true` when its
  `last_post_at` is later than the user's `last_read_at` for that thread
  (NULL last_read_at is treated as the Unix epoch, so never-opened
  threads with activity are unread). Empty threads (`last_post_at IS
  NULL`) are always `has_unread: false`.

  Order matches `list_threads/1`: stickies first, then newest activity.
  Preloads `:created_by` so callers can render the creator handle
  without a follow-up N+1 query.

  When `user_id` is `nil`, delegates to `list_threads/1` and annotates
  every row with `has_unread: false` — the "no user context" case used
  by admin paths and tests that don't care about read-state.
  """
  @spec list_threads(String.t(), String.t() | nil) :: [map()]
  def list_threads(board_id, nil) do
    board_id
    |> list_threads()
    |> Enum.map(&annotate_no_user/1)
  end

  def list_threads(board_id, user_id) when is_binary(user_id) do
    # Single query: threads LEFT JOIN thread_read_pointers for this user.
    # `has_unread` computed inline via a SQL expression so we don't need
    # a second pass in Elixir.
    epoch = ~U[1970-01-01 00:00:00.000000Z]

    query =
      from t in Thread,
        left_join: trp in ReadPointer,
        on: trp.thread_id == t.id and trp.user_id == ^user_id,
        where: t.board_id == ^board_id and is_nil(t.deleted_at),
        order_by: [desc: t.sticky, desc: t.last_post_at],
        preload: [:created_by],
        select: %{
          thread: t,
          has_unread:
            not is_nil(t.last_post_at) and
              t.last_post_at > coalesce(trp.last_read_at, ^epoch)
        }

    query
    |> Repo.all()
    |> Enum.map(fn %{thread: t, has_unread: u} ->
      t
      |> Map.from_struct()
      |> Map.put(:has_unread, u)
    end)
  end

  # Annotate a plain thread struct (from list_threads/1) with a false
  # has_unread field — used when no user_id is supplied so the caller
  # API is uniform (all rows are maps with :has_unread).
  defp annotate_no_user(%Thread{} = t) do
    t
    |> Map.from_struct()
    |> Map.put(:has_unread, false)
  end
```

### 4. Note on `Map.from_struct/1` and the `:created_by` preload

When we call `Map.from_struct(t)` on a preloaded thread, the `:created_by` field contains the loaded `%Foglet.Accounts.User{}` struct (if present) or `%Ecto.Association.NotLoaded{}`. The resulting map retains this association — callers accessing `thread.created_by.handle` from the map still work because `map.key.subkey` works on atom-keyed maps the same as on structs. This is identical to how `list_subscribed_boards/1` returns `%{b | unread_count: n}` (struct updated with a virtual field).

Note: We use `Map.from_struct(t)` rather than `%{t | has_unread: true}` because `:has_unread` is NOT a field on the `Thread` struct — the latter would fail with `KeyError`. Returning plain maps with atom keys is the contract.

    test "returns empty list for a board with no threads" do
      {board, _pid} = setup_board_with_server()

      assert [] = Foglet.Threads.list_threads(board.id, user.id)

      poster = user_fixture()
      reader = user_fixture()

      {:ok, %{thread: t1}} =
        Foglet.Threads.create_thread(board.id, poster.id, %{title: "A", body: "b"})

      {:ok, %{thread: t2}} =
        Foglet.Threads.create_thread(board.id, poster.id, %{title: "B", body: "b"})

      results = Foglet.Threads.list_threads(board.id, reader.id)

      assert length(results) == 2

      result_map = Map.new(results, fn t -> {t.id, t.has_unread} end)
      assert Map.get(result_map, t1.id) == true
      assert Map.get(result_map, t2.id) == true
    end

    test "thread read before its last post is has_unread: true" do
      {board, _pid} = setup_board_with_server()
      poster = user_fixture()
      reader = user_fixture()

      {:ok, %{thread: thread, post: p1}} =
        Foglet.Threads.create_thread(board.id, poster.id, %{title: "T", body: "root"})

      # Reader opens thread and pointer is flushed at the root post
      {:ok, _} = Foglet.Threads.advance_thread_read_pointer(reader.id, thread.id, p1.id)

      # Sleep is forbidden — instead, manually backdate the read pointer so
      # any new post afterwards is guaranteed to be newer.
      past = DateTime.add(DateTime.utc_now(), -3600, :second)

      from(rp in Foglet.Threads.ReadPointer,
        where: rp.user_id == ^reader.id and rp.thread_id == ^thread.id
      )
      |> Repo.update_all(set: [last_read_at: past])

      # Now another user replies — last_post_at updates to ~now
      {:ok, _p2} = Foglet.Posts.create_reply(thread.id, board.id, poster.id, %{body: "later"})

      [%{has_unread: has_unread}] = Foglet.Threads.list_threads(board.id, reader.id)
      assert has_unread == true
    end

    test "thread read up-to-date is has_unread: false" do
      {board, _pid} = setup_board_with_server()
      poster = user_fixture()
      reader = user_fixture()

        Foglet.Threads.create_thread(board.id, poster.id, %{title: "T", body: "root"})

      # We set last_read_at to the future to guarantee the comparison
      # (thread.last_post_at > trp.last_read_at) is false.

        where: rp.user_id == ^reader.id and rp.thread_id == ^thread.id

      [%{has_unread: has_unread}] = Foglet.Threads.list_threads(board.id, reader.id)
      assert has_unread == false
    end

    test "preserves order: stickies first, then newest last_post_at" do

      poster = user_fixture()
      reader = user_fixture()

      {:ok, %{thread: t_old}} =
        Foglet.Threads.create_thread(board.id, poster.id, %{title: "old", body: "b"})

      {:ok, %{thread: t_sticky}} =
        Foglet.Threads.create_thread(board.id, poster.id, %{title: "sticky", body: "b"})

      {:ok, %{thread: t_new}} =
        Foglet.Threads.create_thread(board.id, poster.id, %{title: "new", body: "b"})

      {:ok, _} = Foglet.Threads.sticky_thread(Foglet.Threads.get_thread!(t_sticky.id))

      # Backdate t_old so its last_post_at is older than t_new's

      from(t in Foglet.Threads.Thread, where: t.id == ^t_old.id)
      |> Repo.update_all(set: [last_post_at: past])

      results = Foglet.Threads.list_threads(board.id, reader.id)
      ids = Enum.map(results, & &1.id)

      # sticky first, then new (recent), then old
      assert ids == [t_sticky.id, t_new.id, t_old.id]
